4. Make a function

a) Use the code above to make a function called get_wiki_info that uses a single input url (a Wikipedia URL) and outputs the data frame of the format above.

get_wiki_info <- function(url){
  keys <- c('Full name', 'Political party', 'Children')
  
  wiki <- read_html(url)
  infobox <- wiki %>% 
              html_nodes('table.vcard') %>%
              html_table(fill = TRUE, header = FALSE)
  df_infobox <- as.data.frame(infobox[[1]])
  colnames(df_infobox) <- c('key','value')
  df_infobox <- df_infobox %>% filter(key != '')
  df_infobox[1,1] <- 'Full name'
  
  df_base <- data.frame('key' = keys) 
  df_result <- merge(df_base, df_infobox, by = 'key', all.x = TRUE) # left join to make sure output NA when value is missing
  
  
  df_result <- df_result[match(keys, df_result$key),]  # make sure output has the order of 'Full name', 'Political party', and 'Children'
  rownames(df_result) <- NULL
  
  df_result
}
